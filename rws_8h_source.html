77  class SHORTLEX : public ReductionOrdering {
78  public:
81  SHORTLEX()
82  : ReductionOrdering([](rws_word_t const* p, rws_word_t const* q) {
83  return (p->size() > q->size()
84  || (p->size() == q->size() && *p > *q));
85  }) {}
86 
92  explicit SHORTLEX(std::function<bool(rws_letter_t const&,
93  rws_letter_t const&)> letter_order)
94  : ReductionOrdering([this](rws_word_t const* p, rws_word_t const* q) {
95  // FIXME This is unsafe since we don't check that p and q consist of
96  // the correct letters
97  if (p->size() > q->size()) {
98  return true;
99  } else if (p->size() < q->size()) {
100  return false;
101  }
102  auto itp = p->cbegin();
103  auto itq = q->cbegin();
104  while (itp < p->cend() && *itp == *itq) {
105  itp++;
106  itq++;
107  }
108  return (itp != p->cend() && this->_letter_order(*itp, *itq));
109  }),
110  _letter_order(letter_order) {}
111 
112  private:
113  std::function<bool(rws_letter_t const&, rws_letter_t const&)> _letter_order;
114  };

433  // Construct from RWS, and pointers to libsemigroups::rws_word_t.
434  //
435  // A rule is guaranteed to have its left hand side greater than its right
436  // hand side according to the reduction ordering of \p rws. The Rule
437  // object constructed here takes ownership of \p p and \p q and deletes
438  // them in its destructor.
439  Rule(RWS const* rws, rws_word_t* p, rws_word_t* q)
440  : _rws(rws), _lhs(p), _rhs(q), _active(false) {
441  assert(*_lhs != *_rhs);
442  reorder();
443  }
444 
445  // Construct from RWS with new but empty rws_word_t's
446  explicit Rule(RWS const* rws)
447  : _rws(rws),
448  _lhs(new rws_word_t()),
449  _rhs(new rws_word_t()),
450  _active(false) {}
451 
452  // Destructor, deletes pointers used to create the rule.
453  ~Rule() {
454  delete _lhs;
455  delete _rhs;
456  }
457 
458  void rewrite() {
459  _rws->rewrite(_lhs);
460  _rws->rewrite(_rhs);
461  reorder();
462  }
463 
464  void rewrite_rhs() {
465  _rws->rewrite(_rhs);
466  }
467 
468  void clear() {
469  _lhs->clear();
470  _rhs->clear();
471  }
472 
473  inline bool is_active() const {
474  return _active;
475  }
476 
477  void deactivate() {
478  _active = false;
479  }
480 
481  void activate() {
482  _active = true;
483  }
484 
485  void reorder() {
486  if ((*(_rws->_order))(_rhs, _lhs)) {
487  std::swap(_lhs, _rhs);
488  }
489  }
490 
491  RWS const* _rws;
492  rws_word_t* _lhs;
493  rws_word_t* _rhs;
494  bool _active;
495  };
