74  class SHORTLEX : public RO {
75  public:
78  SHORTLEX()
79  : RO([](rws_word_t const& p, rws_word_t const& q) {
80  return (p.size() > q.size() || (p.size() == q.size() && p > q));
81  }) {}
82 
88  explicit SHORTLEX(std::function<bool(rws_letter_t const&,
89  rws_letter_t const&)> letter_order)
90  : RO([this](rws_word_t const& p, rws_word_t const& q) {
91  // FIXME This is unsafe since we don't check that p and q consist of
92  // the correct letters
93  if (p.size() > q.size()) {
94  return true;
95  } else if (p.size() < q.size()) {
96  return false;
97  }
98  auto itp = p.cbegin();
99  auto itq = q.cbegin();
100  while (itp < p.cend() && *itp == *itq) {
101  itp++;
102  itq++;
103  }
104  return (itp != p.cend() && this->_letter_order(*itp, *itq));
105  }),
106  _letter_order(letter_order) {}
107 
108  private:
109  std::function<bool(rws_letter_t const&, rws_letter_t const&)> _letter_order;
110  };
