530  size_t crank() const {
531  _lookup.clear();
532  _lookup.resize(degree(), false);
533  size_t r = 0;
534  for (auto const& x : *(this->_vector)) {
535  if (x != UNDEFINED && !_lookup[x]) {
536  _lookup[x] = true;
537  r++;
538  }
539  }
540  return r;

546  void cache_hash_value() const override {
547  size_t seed = 0;
548  size_t deg = this->degree();
549  for (auto const& val : *(this->_vector)) {
550  seed *= deg;
551  seed += val;
552  }
553  this->_hash_value = seed;
554  }

562  std::vector<S>* vector(new std::vector<S>());
563  vector->reserve(this->degree());
564  for (size_t i = 0; i < this->degree(); i++) {
565  vector->push_back(i);
566  }
567  return new T(vector);
568  }

576  private:

578  static std::vector<bool> _lookup;

678  class PartialPerm : public PartialTransformation<T, PartialPerm<T>> {
679  public:
690  explicit PartialPerm(std::vector<T>* vector, size_t hv = Element::UNDEFINED)
691  : PartialTransformation<T, PartialPerm<T>>(vector, hv) {}
692 
697  explicit PartialPerm(std::vector<T> const& vector)
698  : PartialTransformation<T, PartialPerm<T>>(vector) {}
699 
708  explicit PartialPerm(std::vector<T> const& dom,
709  std::vector<T> const& ran,
710  size_t deg)
711  : PartialTransformation<T, PartialPerm<T>>() {
712  LIBSEMIGROUPS_ASSERT(dom.size() == ran.size());
713  LIBSEMIGROUPS_ASSERT(dom.empty()
714  || deg >= *std::max_element(dom.begin(), dom.end()));
715 
716  this->_vector->resize(deg + 1, PP_UNDEFINED);
717  for (size_t i = 0; i < dom.size(); i++) {
718  (*this->_vector)[dom[i]] = ran[i];
719  }
720  }
721 
729  bool operator<(const Element& that) const override {
730  auto pp_this = static_cast<const PartialPerm<T>*>(this);
731  auto pp_that = static_cast<const PartialPerm<T>&>(that);
732 
733  size_t deg_this = pp_this->degree();
734  for (auto it = pp_this->_vector->end() - 1;
735  it >= pp_this->_vector->begin();
736  it--) {
737  if (*it == PP_UNDEFINED) {
738  deg_this--;
739  } else {
740  break;
741  }
742  }
743  size_t deg_that = pp_that.degree();
744  for (auto it = pp_that._vector->end() - 1;
745  it >= pp_that._vector->begin() && deg_that >= deg_this;
746  it--) {
747  if (*it == PP_UNDEFINED) {
748  deg_that--;
749  } else {
750  break;
751  }
752  }
753 
754  if (deg_this != deg_that) {
755  return deg_this < deg_that;
756  }
757 
758  for (size_t i = 0; i < deg_this; i++) {
759  if ((*pp_this)[i] != pp_that[i]) {
760  return (*pp_this)[i] == PP_UNDEFINED
761  || (pp_that[i] != PP_UNDEFINED && (*pp_this)[i] < pp_that[i]);
762  }
763  }
764  return false;
765  }
766 
773  Element* really_copy(size_t increase_deg_by = 0) const override {
774  std::vector<T>* vector_copy = new std::vector<T>(*this->_vector);
775  if (increase_deg_by == 0) {
776  return new PartialPerm<T>(vector_copy, this->_hash_value);
777  } else {
778  size_t n = vector_copy->size();
779  vector_copy->reserve(n + increase_deg_by);
780  for (size_t i = n; i < n + increase_deg_by; i++) {
781  vector_copy->push_back(PP_UNDEFINED);
782  }
783  return new PartialPerm<T>(vector_copy);
784  }
785  }
