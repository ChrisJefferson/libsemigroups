277  class NaturalSemiring : public SemiringWithThreshold {
278  public:
285  NaturalSemiring(int64_t t, int64_t p)
286  : SemiringWithThreshold(t), _period(p) {
287  assert(_period > 0);
288  assert(this->threshold() > 0);
289  }
290 
292  int64_t one() const override {
293  return 1;
294  }
295 
297  int64_t zero() const override {
298  return 0;
299  }
300 
303  int64_t prod(int64_t x, int64_t y) const override {
304  assert(x >= 0 && x <= _period + this->threshold() - 1);
305  assert(y >= 0 && y <= _period + this->threshold() - 1);
306  return thresholdperiod(x * y);
307  }
308 
311  int64_t plus(int64_t x, int64_t y) const override {
312  assert(x >= 0 && x <= _period + this->threshold() - 1);
313  assert(y >= 0 && y <= _period + this->threshold() - 1);
314  return thresholdperiod(x + y);
315  }
316 
318  int64_t period() const {
319  return _period;
320  }
321 
322  private:
323  int64_t thresholdperiod(int64_t x) const {
324  int64_t threshold = this->threshold();
325  if (x > threshold) {
326  return threshold + (x - threshold) % _period;
327  }
328  return x;
329  }
330 
331  int64_t _period;
332  };
